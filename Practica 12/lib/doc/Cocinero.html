def degustar_uno()
	if @contador == 0
		if @annos > 35
			return "El cocinero se ha jubilado y no hay mas platos disponibles"
		else
			return "No hay platos disponibles"
		end
	else
		@contador -= 1
		return "Que delicioso está el plato"
	end
end

def uno_mas()
	@annos += 1
	if @annos > 3
		@experiencia = @annos * 2
		@contador += @experiencia * 1	
	end 
end
